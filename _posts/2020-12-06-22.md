# 난독화 전 C코드

```c
#include <stdio.h>

int main()
{
  int temp = 10;
  int i;

  for (i=0; i<10; i++)
  {
    if(temp < 15){
      temp = temp + 1;
    }
    else{
      temp = temp - 1;
    }
  }

  if(temp==14){
    temp = temp*100;
  }

  printf("%d\n", temp);
}

```
